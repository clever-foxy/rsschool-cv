Kata from CODEWARS: Heads and Legs. Everybody has probably heard of the animal heads and legs problem. Assuming there are no other animal species besides cows and chickens, count how many of each animal are there. 
                    function animals(heads, legs){

                        if (heads === 0 && legs === 0){
                          return [0, 0];
                        }
                        if (heads <= 0 || legs <= 0 ){
                          return "No solutions";
                        }
                        let countCows = legs/2 - heads,
                            countChickens = heads - countCows;
                        if (countChickens < 0 || countCows < 0) {
                          return "No solutions";
                        }
                        if ( countCows % 1 !== 0 || countChickens % 1 !== 0 ) {
                          return "No solutions";
                        }
                        return [countChickens, countCows];
                      
                      }
